#)Code For Equilibration of LJ Fluid
units lj
atom_style sphere
lattice sc 1.0
region box block 0 30 0 30 0 30 units box
create_box 2 box
create_atoms 1 box
set type 1 diameter 1.0
velocity all create 1.0 87287 loop geom
# multi neighbor and comm for efficiency
neighbor 1 multi
neigh_modify delay 0
comm_modify mode multi
# colloid potential
pair_style lj/cut 2.5 
pair_coeff * * 1.0 1.0 2.5
fix 1 all nvt temp 1.0 1.0 1.0
fix 2 all langevin 1.0 1.0 0.1 437624
dump 1 all atom 1000 dump.colloid
thermo_style custom step temp epair etotal press vol density
thermo 1000
timestep 0.005
run 50000 
reset_timestep 0
unfix 1
change_box all z final 0.0 80.0 boundary p p f units box 
fix 1 all nve
dump 2 all atom 1000 dump.equilibrium 
thermo_modify lost ignore flush yes
run 200000
write_data data.equilibrium
